NON-TERMINALS

Statement ::= Compound_stmt
| Assignment
| FunctionDeclaration
| If_stmt
Compound_stmt ::= <LEFT_BRACE> Statement_list <RIGHT_BRACE>
Statement_list ::= ( Statement )*
If_stmt ::= If_part Else_part
If_part ::= <IF> <LEFT_PARAN> Expression <RIGHT_PARAN> Statement
Else_part ::= <ELSE> Statement
ArgumentList ::= <LEFT_PARAN> ( IdentifierType ( <COMMA> IdentifierType )* )? <RIGHT_PARAN>
PureFunction ::= <LEFT_BRACE> Statement_list <RETURN> Expression <SEMICOLON> <RIGHT_BRACE>
SideEffectFunction ::= <LEFT_BRACE> Statement_list ( <RETURN> <SEMICOLON> )? <RIGHT_BRACE>
FunctionDeclaration ::= <FUNCTION> identifier ArgumentList ( <COLON> ( typeSignature PureFunction | <VOID> SideEffectFunction ) | SideEffectFunction )
Assignment ::= VarIdentifierType ( <EQ> Expression )? <SEMICOLON>
| identifier <EQ> Expression <SEMICOLON>
VarIdentifierType ::= <VAR> IdentifierType
IdentifierType ::= identifier <COLON> typeSignature
typeSignature ::= <NUMBER>
| <BOOLEAN>
| <STRING> 
// all the Expression
Expression ::= SimpleExpression ( ( <DOUBLE_EQ> | <NOT_EQ> | <LT> | <LE> | <GT> | <GE> ) SimpleExpression )?
SimpleExpression ::= ( ( <PLUS> | <MINUS> )? term ( ( <PLUS> | <MINUS> | <OR> ) term )* )
| term
term ::= ( factor ( ( <MUL> | <DIV> | <MOD> | <AND> ) factor )* )
| factor
factor ::= <NOT> factor
| identifier
| number
| string
| <LEFT_PARAN> Expression <RIGHT_PARAN>
identifier ::= <IDENTIFIER>
number ::= <NUM>
string ::= <STRING_LITERAL>
